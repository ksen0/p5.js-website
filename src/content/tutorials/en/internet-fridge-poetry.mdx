function shapeText(word, x, y) {
  
  let contours = font.textToContours(word, x, y, { sampleFactor: 0.03 })
  
  beginShape()
  for (const points of contours) {
    beginContour()
    for (let p of points) {
      vertex(p.x, p.y)
    }
    endContour(CLOSE)
  }
  endShape()
  
}
